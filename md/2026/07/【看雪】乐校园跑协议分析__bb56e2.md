> 
> - **登录加密机制**：登录请求使用AES/ECB加密，密钥基于时间戳动态生成，算法实现在native层so文件中。
> - **密钥生成算法**：从时间戳提取第3-5位和4-7位字符，计算差值后左移最后一位数字，拼接固定字符串取前16位作为密钥。

    var redirect = function (args, pathIndex) {
        var path = ptr(args[pathIndex]).readCString()
        if (path && path.indexOf("/maps") !== -1 && path.indexOf("proc") !== -1) {
            console.log("[*] Detected maps check: " + path);
            console.log("    -> Redirecting to: " + fakeMaps);
            var fakePathPtr = Memory.allocUtf8String(fakeMaps);
            args[pathIndex] = fakePathPtr;
        }
    }
    if (open) {
        Interceptor.attach(open, {
            onEnter: function (args) {
                redirect(args, 0);
            }
        });
    }
    if (openat) {
        Interceptor.attach(openat, {
            onEnter: function (args) {
                redirect(args, 1);
            }
        });
    }
}
